git init - Initializes the current folder as a git repository
git status - current status of the folder
git add - add file(s)/folder(s) to staging area
#staging area - snapshot of file(s)/folder(s) ready to commit to the repository
git commit -m "This is a description for the commit." - commit the changes to the git repository
git log --oneline - logs of commits being put into git repository
git checkout <commit> <file> - checkout the file from an older commit
git push -u origin master - Push the local git repository to the origin of the master branch
git clone <repository URL> - Clone an online Git repository to your computer
